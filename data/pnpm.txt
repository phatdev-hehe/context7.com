async function writeLockfiles(lockfile, opts) {
  const { fileName, throwOnConvertionProblems } = opts
  const dir = opts.dir ?? opts.lockfileDir
  const normalizedLockfile = normalizeLockfile(lockfile)
  const yaml = stringifyLockfile(normalizedLockfile)
  await writeFileAtomically(path.join(dir, fileName), yaml)
}

TITLE: Major Version Change Documentation
DESCRIPTION: Documentation for version 2.0.0 that introduces changes to store index file storage using content hash and package identifier to handle identical content published under different packages or versions.

LANGUAGE: markdown
CODE:
### Major Changes

- d433cb9: Some registries allow identical content to be published under different package names or versions. To accommodate this, index files in the store are now stored using both the content hash and package identifier.

  This approach ensures that we can:

  1. Validate that the integrity in the lockfile corresponds to the correct package,
     which might not be the case after a poorly resolved Git conflict.
  2. Allow the same content to be referenced by different packages or different versions of the same package.

  Related PR: [#8510](https://github.com/pnpm/pnpm/pull/8510)
  Related issue: [#8204](https://github.com/pnpm/pnpm/issues/8204)
